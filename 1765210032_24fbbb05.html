// JavaScript code to highlight differences and enable synchronized scrolling
document.addEventListener('DOMContentLoaded', function() {
  const contract1 = document.getElementById('contract1');
  const contract2 = document.getElementById('contract2');
  const highlightAdd = 'highlight-add';
  const highlightRemove = 'highlight-remove';
  const highlightChange = 'highlight-change';

  // Function to highlight differences
  function highlightDifferences() {
    const lines1 = contract1.innerText.split('
');
    const lines2 = contract2.innerText.split('
');
    let i = 0;

    // Loop through lines and highlight differences
    while (i < lines1.length && i < lines2.length) {
      const line1 = lines1[i];
      const line2 = lines2[i];

      if (line1 !== line2) {
        // Check if the line is added, removed, or changed
        if (!line2.includes(line1)) {
          // Line is removed
          line1.classList.add(highlightRemove);
        } else if (!line1.includes(line2)) {
          // Line is added
          line1.classList.add(highlightAdd);
        } else {
          // Line is changed
          line1.classList.add(highlightChange);
        }
      }

      i++;
    }
  }

  // Enable synchronized scrolling
  let isScrolling;
  const scrollAmount = 5; // Adjust as needed

  contract1.addEventListener('scroll', function() {
    clearTimeout(isScrolling);
    isScrolling = setTimeout(function() {
      contract2.scrollTop = contract1.scrollTop;
    }, scrollAmount);
  });

  contract2.addEventListener('scroll', function() {
    clearTimeout(isScrolling);
    isScrolling = setTimeout(function() {
      contract1.scrollTop = contract2.scrollTop;
    }, scrollAmount);
  });

  // Highlight differences on initial load
  highlightDifferences();
});
